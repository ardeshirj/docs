Atom Fedora-20
--------------

# This assumes that you installed npm
# otherwise read npm-install.txt first

# Install dependencies
  sudo yum install 
    nodejs 
    libgnome-keyring-devel

# Remove conflict
  sudo yum remove gyp

# Clone atom form github
  git clone https://github.com/atom/atom
  cd atom

# Creates application at $TMPDIR/atom-build/Atom
  script/build

# Run install script
  sudo script/grunt install

# It installs it at /usr/local/share/atom
# and creates atom/apm executable in /usr/local/bin
# also create .desktop at /usr/local/share/applications

# Re-install gyp
  sudo yum install gyp
