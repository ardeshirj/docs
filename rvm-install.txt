# General Setup

  rvm install 2.1.2 # Install ruby verison 2.1.2
  rvm list # List all installed ruby versions

  rvm docs generate # Generate ri docs for current ruby version
  # Generate docs for other rubyO versions that are installed:
    rvm cleanup all

# Gemset
  rvm gemset create [you-name-it] # Create new gemset to use for a project
  rvm gemset list # List all the existing gemset for the current rubverison
  rvm use gemset [you-name-it] # Use the new gemset

  # Reset the gemset to default (SKIP this if you install gem via bundle)
    rvm gemset use [gemset-you-want-to-delete]
    rvm gemset empty

# Bundle-Gemfile setup
  bundle init # Generate Gemfile by using gem called bundle
  vim Gemfile # Add any gem files that needed for your project for example:
    gem 'rails', '~> 4.2.0'
    gem 'travis', '~> 1.7.4'
  bundle install # Download and install gems and create Gemfile.lock
  bundle show OR gem list # To list installed gem in the gemset
  bundle clean --force # Will remove gems that is not in Gemfile anymore

# To set ruby and gemset for the project (this will create .versions.conf)
  rvm --create --versions-conf use [ruby-version]@[the-gemset]

# Remove ruby or gemset
  rvm gemset remove [gemset-name] # Remove gemset
  rvm remove [ruby-version] # Remove specefic ruby version
