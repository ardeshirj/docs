# General Setup
  rvm list known # Show most recent ruby versions to install
  rvm install 2.1.2 # Install ruby verison 2.1.2
  rvm list # List all installed ruby versions
  rvm docs generate # Generate ri docs for current ruby version
    # Generate docs for other ruby versions that are installed:
    rvm cleanup all
    rvm all do rvm docs generate
  rvm gemset create [you-name-it] # Create new gemset to use for a project
  rvm gemset list # List all the existing gemset for the current rubverison
  rvm use gemset [you-name-it] # Use the new gemset

# Rails setup
  gem install rails # Using rvm gem to install gems in gemset
  rails new [you-app-name] # Create rails structure. Also run bundle install

# Non-rails setup 
  bundle init or install # Generate or install Gemfile by using gem called bundle
  vim Gemfile # Add any gem files that needed for your project
  bundle install # Download and install gems and create Gemfile.lock
  # In order to clean up the installed gems:
    vim Gemfile  # Delete all gems you want to delete
    bundle clean --force # Will delete previously installed by bundle

# To set ruby and gemset for the project 
  rvm --create --versions-conf use [ruby-version]@[the-gemset] # Create .versions.conf ofr

# rvm download folder
  ~/.rvm/archives/
 
# Update rvm
  rvm get stable

# To remove
  rvm implode
