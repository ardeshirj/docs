NPM Fedora-20 
-------------

# *** USE nvm INSTEAD: nvm comes with npm***

# Download install sh
  https://www.npmjs.org/install.sh

# Use nodejs-install.txt to install nodejs

# Install it in /some/path
  npm_config_prefix=/some/path sh install.sh
    [some/path = ~/Development/npm]

# Set the paths for global packages
# and add the packages bin folder to path
  vim ~/.npmrc
    prefix = ~/Development/npm/global
  vim ~/.bashrc
    export PATH=$PATH:~/Development/npm/bin
    export PATH=$PATH:~/Development/npm/global/bin

# To remove:
  rm -rf ~/Development/npm
  sudo yum remove nodejs
  # if  previously installed by yum
    sudo rm -rf /usr/lib/node_modules
