NVM Fedora-20
----------------

# To install:
  curl https://raw.githubusercontent.com/creationix/nvm/v0.17.3/install.sh | bash

# Add these to lines to .profile to be consistent with rvm
  export NVM_DIR="/home/floyd/.nvm"
  [ -s "$NVM_DIR/nvm.sh" ] && . "$NVM_DIR/nvm.sh"  # This loads nvm

# To activate nvm add the following line to .bashrc
  source ~/.nvm/nvm.sh

# To get list of nodejs version to install
  nvm ls-remote

# Use the alias 'default' to have node in shell
  nvm alias default 0.10.33 [stable]

# Default path for npm to install package globally:
  ~/.nvm/versions/[node-version]/lib

# To check everything install correctly
  nvm
  node -v

# To update
  For now, just install it again

# To remove
  rm -rf ~/.nvm
  # Also remove all lines in ~/.bash_profile, ~/.zshrc or ~/.profile
